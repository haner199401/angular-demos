ng-init 初始化 {{msg}}

    启动方式1:
    ng-app="" 启动 angular

    启动方式2:
    angular.module('ng-app的名称');

    启动方式3:
    angular.module('myApp', []).controller('MyController', ['$scope', function ($scope) {
        $scope.msg = 'hello,world';
    }]);

    angular.element(document).ready(function() {
        angular.bootstrap(document, ['myApp']);
    });
